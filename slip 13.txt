// number of words and Lines in that file.

import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
class Q1{
public static void main(String args[]) throws Exception{
// FileInputStream in = new FileInputStream(args[0]);
BufferedReader br = new BufferedReader(new FileReader(args[0]));

String str;

int lines=0,wordsCount=0;

File f= new File(args[0]);

if(f.exists()){

while((str = br.readLine())!=null){

lines++;

String words[] = str.trim().split("\\s+");

wordsCount += words.length;

}

System.out.println("\nNo of Lines: "+lines);

System.out.println("\nNo of Words: "+wordsCount);

}

}

}














import java.time.LocalDateTime;
import java.time.ZonedDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Locale;

class Q2 {
    public static void main(String[] args) {
        LocalDateTime dateTime = LocalDateTime.now();
        
        ZonedDateTime zonedDateTime = ZonedDateTime.now(ZoneId.of("Asia/Kolkata"));

        DateTimeFormatter f1 = DateTimeFormatter.ofPattern("dd/MM/yyyy");
        System.out.println("Current date is : " + dateTime.format(f1));

        DateTimeFormatter f2 = DateTimeFormatter.ofPattern("MM-dd-yyyy");
        System.out.println("Current date is : " + dateTime.format(f2));

        DateTimeFormatter f3 = DateTimeFormatter.ofPattern("EEEE MMMM dd yyyy", Locale.ENGLISH);
        System.out.println("Current date is : " + dateTime.format(f3));

        DateTimeFormatter f4 = DateTimeFormatter.ofPattern("EEE MMMM dd HH:mm:ss z yyyy", Locale.ENGLISH);
        System.out.println("Current date and time is : " + zonedDateTime.format(f4));

        DateTimeFormatter f5 = DateTimeFormatter.ofPattern("dd/MM/yy HH:mm:ss a Z");
        System.out.println("Current date and time is : " + zonedDateTime.format(f5));
    }
}
