class Item{
    int ino;
    String iname;
    double iprice;
    static int count=0;

    Item(int ino,String iname,double iprice){
        this.ino = ino;
        this.iname = iname;
        this.iprice = iprice;
        count++;
    }
    static void getCount(){
        System.out.println("\nObjects created: "+count);
    }
    void disp(){
        System.out.println("\nItem Name: "+iname+"\nItem No: "+ino+"\nItem Price: "+iprice);
    }
}

import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.PrintStream;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.time.temporal.ChronoUnit;
import java.util.Scanner;
class Donor{
    String name,add,cno,bgroup,dold;
    int age;

    Donor(String name, String add, String cno, String bgroup, String dold, int age) {
        this.name = name;
        this.add = add;
        this.cno = cno;
        this.bgroup = bgroup;
        this.dold = dold;
        this.age = age;
    }
    boolean isEligible(){
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("dd-MM-yyyy");
        LocalDate lastDonationDate = LocalDate.parse(dold,formatter);
        LocalDate currentDate = LocalDate.now();
        long months = ChronoUnit.MONTHS.between(lastDonationDate,currentDate);
        if(months>6){
            return true;
        }
        return false;
    }
    
}
