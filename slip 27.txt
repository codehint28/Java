class Employee{
    int salary =50000;
    int getSalary(){
        return salary;
    }
}
class Manager extends Employee{
    int traveling = 5000,rent = 10000;
    int getSalary(){
        return salary+traveling+rent;
    }
}
class Q1{
    public static void main(String args[]){
        Employee e = new Employee();
        Manager m = new Manager();

        System.out.println("\nEmployee Salary: "+e.getSalary());
        System.out.println("\nManager Salary: "+m.getSalary());
    }
}








import java.io.File;
import java.util.Scanner;
public class Q2 {
    public static void main(String args[]){
        String fname = args[0];
        File fp = new File(fname);
        String confirm;
        Scanner sc = new Scanner(System.in);

        if(fp.isDirectory()){
            int count =0;
            String files[] = fp.list();

            if(files.length>0){
                System.out.print("\nDo you want to Delete all Files From this Directory? (y/n): ");
                confirm = sc.next();

                if(confirm.equals("y")){

                    for(int i=0;i<files.length;i++){
                        File f = new File(fname+"/"+files[i]);
                        f.delete();
                        count++;
                    }
                }
            }
            
            System.out.println(" "+count+" Files Deleted Succesfully");
        }

        else if(fp.isFile()){
            System.out.println("Name of File: "+fp.getName());
            System.out.println("Last Modified : "+fp.lastModified());
            System.out.println("Absolute Path: "+fp.getAbsolutePath());
        }
    }    
}
