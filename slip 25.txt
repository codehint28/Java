import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
class Student{
    int roll;
    String name,cls;
    double per;

    Student(int roll,String name,String cls,double per){
        this.roll = roll;
        this.name = name;
        this.cls = cls;
        this.per = per;
    }
    void disp(){
        System.out.println("\nName: "+name+"\nRoll no: "+roll+"\nClass: "+cls+"\nPercentage: "+per);
    }
}
class Q1{
    public static void main(String args[]) throws IOException {    
        int roll;
        String name,cls;
        double per;

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.print("\nEnter the name: ");
        name = br.readLine();
        System.out.print("\nEnter roll no: ");
        roll = Integer.parseInt(br.readLine());
        System.out.print("\nEnter the Class: ");
        cls = br.readLine();
        System.out.print("\nEnter the Percentage: ");
        per = Double.parseDouble(br.readLine());

        Student s = new Student(roll, name, cls, per);
        s.disp();
    }
}
